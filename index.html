<?php

    $name = $conn->real_escape_string($_POST['name']);
    $age = intval($_POST['age']); // Ensure age is an integer
    $ailment = $conn->real_escape_string($_POST['ailment']);

    $sql = "INSERT INTO patients (name, age, ailment) VALUES ('$name', $age, '$ailment')";

edit.php (Update Record) 
<?php
include 'db.php';

// Check for ID parameter
if (!isset($_GET['id'])) {
    header("Location: index.php");
    exit();
}

$id = intval($_GET['id']);
$sql = "SELECT id, name, age, ailment FROM patients WHERE id = $id";
$result = $conn->query($sql);
$patient = $result->fetch_assoc();

// If POST request, handle update
if ($_SERVER["REQUEST_METHOD"] == "POST" && $patient) {
    $name = $conn->real_escape_string($_POST['name']);
    $age = intval($_POST['age']);
    $ailment = $conn->real_escape_string($_POST['ailment']);
    
    $sql = "UPDATE patients SET name = '$name', age = $age, ailment = '$ailment' WHERE id = $id";
    
    if ($conn->query($sql) === TRUE) {
        header("Location: index.php");
        exit();
    } else {
        $error = "Error: " . $conn->error;
    }
} else if (!$patient) {
    // Redirect if patient not found
    header("Location: index.php");
    exit();
}

?>
<!DOCTYPE html>
<html>
<head>
    <title>Edit Patient - VitClinic</title>
    <link rel="stylesheet" href="style.css">
</head>
<body>
    <div class="container">
        <h1>Edit Patient</h1>
        
        <?php if (isset($error)) echo "<p class='error'>$error</p>"; ?>
        
        <form method="POST">
            <label for="name">Patient Name:</label>
            <input type="text" name="name" value="<?php echo htmlspecialchars($patient['name']); ?>" required>
            
            <label for="age">Age:</label>
            <input type="number" name="age" value="<?php echo $patient['age']; ?>" required>
            
            <label for="ailment">Ailment:</label>
            <textarea name="ailment" required><?php echo htmlspecialchars($patient['ailment']); ?></textarea>
            
            <button type="submit">Update Patient</button>
            <a href="index.php" class="btn-cancel">Cancel</a>
        </form>
    </div>
</body>
</html>
<?php
$conn->close();
?>

Database Setup (MySQL via XAMPP CMD)

Database Credentials

These credentials are used in the db.php file.

Database Name: vitclinic_db
Username: root
Password: (empty/none)

Step-by-Step XAMPP Setup Instructions

For Windows:
  1. Open Command Prompt and navigate to the XAMPP MySQL directory:
    cd C:\xampp\mysql\bin
  2. Log in to MySQL Monitor using the default root account:
    mysql -u root -p
    Press Enter when prompted for password (leave it blank).
  3. Create the database:
    CREATE DATABASE vitclinic_db;
  4. Switch to the new database:
    USE vitclinic_db;
  5. Create the patients table:
    CREATE TABLE patients (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(100) NOT NULL, age INT, ailment VARCHAR(255));
  6. Verify table creation:
    SHOW TABLES;
  7. Exit MySQL Monitor:
    EXIT;
For Mac/Linux:
  1. Open Terminal and navigate to the XAMPP MySQL directory (or ensure MySQL is in PATH):
    # For XAMPP on Mac: cd /Applications/XAMPP/xamppfiles/bin
  2. Log in to MySQL:
    ./mysql -u root -p
    Press Enter when prompted for password.
  3. Follow steps 3-7 above (same SQL commands).

Final Steps

After creating the database and table:

  1. Place all PHP files (db.php, index.php, add.php, edit.php, delete.php, style.css) in your XAMPP htdocs directory (e.g., C:\xampp\htdocs\vitclinic)
  2. Start XAMPP and ensure MySQL and Apache are running
  3. Visit http://localhost/vitclinic/index.php in your browser
  4. Your VitClinic CRUD application should now be ready to use!
